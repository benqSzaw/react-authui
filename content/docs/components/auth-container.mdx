    Copy and paste the following code into your project.

```tsx title="auth-container.tsx"
import { ComponentProps, ReactNode } from 'react';
import { cn } from '@/lib/utils';
import {
  Card,
  CardContent,
  CardDescription,
  CardFooter,
  CardHeader,
  CardTitle,
} from '@/components/ui/card';

type Props = {
  title: string;
  description?: string;
  footer?: ReactNode;
  children: ReactNode;
} & ComponentProps<'section'>;

function AuthContainer(props: Props) {
  const { className, title, description, footer, children, ...rest } = props;

  return (
    <section
      className={cn(
        'mx-auto flex size-full max-w-lg items-center justify-center p-4',
        className,
      )}
      {...rest}
    >
      <Card className="w-full">
        <CardHeader>
          <CardTitle>
            <h1>{title}</h1>
          </CardTitle>
          {description && (
            <CardDescription>
              <p>{description}</p>
            </CardDescription>
          )}
        </CardHeader>
        <CardContent>{children}</CardContent>
        {footer && (
          <CardFooter>
            <p>{footer}</p>
          </CardFooter>
        )}
      </Card>
    </section>
  );
}

export { AuthContainer };

```

    Update the import paths to match your project setup.
